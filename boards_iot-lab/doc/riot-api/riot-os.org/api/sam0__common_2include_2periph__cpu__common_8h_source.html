    326 static inline int sercom_id(void *sercom)
    327 {
    328 #if defined(CPU_FAM_SAMD21)
    329  return ((((uint32_t)sercom) >> 10) & 0x7) - 2;
    330 #elif defined (CPU_FAM_SAML10) || defined (CPU_FAM_SAML11)
    331  return ((((uint32_t)sercom) >> 10) & 0x7) - 1;
    332 #elif defined(CPU_FAM_SAML21) || defined(CPU_FAM_SAMR30)
    333  /* Left side handles SERCOM0-4 while right side handles unaligned address of SERCOM5 */
    334  return ((((uint32_t)sercom) >> 10) & 0x7) + ((((uint32_t)sercom) >> 22) & 0x04);
    335 #endif
    336 }

    343 static inline void sercom_clk_en(void *sercom)
    344 {
    345 #if defined(CPU_FAM_SAMD21)
    346  PM->APBCMASK.reg |= (PM_APBCMASK_SERCOM0 << sercom_id(sercom));
    347 #else
    348  if (sercom_id(sercom) < 5) {
    349  MCLK->APBCMASK.reg |= (MCLK_APBCMASK_SERCOM0 << sercom_id(sercom));
    350  }
    351 #if defined(CPU_FAM_SAML21)
    352  else {
    353  MCLK->APBDMASK.reg |= (MCLK_APBDMASK_SERCOM5);
    354  }
    355 #endif /* CPU_FAM_SAML21 */
    356 #endif
    357 }

    364 static inline void sercom_clk_dis(void *sercom)
    365 {
    366 #if defined(CPU_FAM_SAMD21)
    367  PM->APBCMASK.reg &= ~(PM_APBCMASK_SERCOM0 << sercom_id(sercom));
    368 #else

    370  MCLK->APBCMASK.reg &= ~(MCLK_APBCMASK_SERCOM0 << sercom_id(sercom));
    371  }
    372 #if defined (CPU_FAM_SAML21)
    373  else {
    374  MCLK->APBDMASK.reg &= ~(MCLK_APBDMASK_SERCOM5);

    376 #endif /* CPU_FAM_SAML21 */
    377 #endif

    386 static inline void sercom_set_gen(void *sercom, uint32_t gclk)
    387 {
    388 #if defined(CPU_FAM_SAMD21)
    389  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN | gclk |
    390  (SERCOM0_GCLK_ID_CORE + sercom_id(sercom)));
    391  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
    392 #else
    393  if (sercom_id(sercom) < 5) {
    394  GCLK->PCHCTRL[SERCOM0_GCLK_ID_CORE + sercom_id(sercom)].reg =
    395  (GCLK_PCHCTRL_CHEN | gclk);
    396  }
    397 #if defined(CPU_FAM_SAML21)
    398  else {
    399  GCLK->PCHCTRL[SERCOM5_GCLK_ID_CORE].reg =
    400  (GCLK_PCHCTRL_CHEN | gclk);
    401  }
    402 #endif /* CPU_FAM_SAML21 */
    403 #endif
    404 }
